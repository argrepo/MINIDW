INSERT INTO IL_Date
(
Date_Val, 
Day_Of_Month, 
Day_Name, 
Day_Of_week, 
Week_Of_Year, 
Calendar_Month, 
Calendar_Month_Name, 
Calendar_Year, 
Calendar_Yrmo, 
Calendar_Quarter, 
Fiscal_Month, 
Fiscal_Yrmo, 
Fiscal_Quarter, 
Fiscal_Year

,IsActive
,Added_Date 
,Added_User
,Updated_Date
,Updated_User
)

Select 

 dates.DatesRange  as  Date_Val
,DAY(dates.DatesRange)  as  Day_Of_Month
,dayname(dates.DatesRange)  as  Day_Name
,DayOfweek(dates.DatesRange)  as  Day_Of_week
,week(dates.DatesRange)  as  Week_Of_Year

,month(dates.DatesRange)  as  Calendar_Month
,monthname(dates.DatesRange)  as  Calendar_Month_Name
,year(dates.DatesRange)  as  Calendar_Year
,EXTRACT(YEAR_MONTH FROM dates.DatesRange)  as  Calendar_Yrmo
,quarter(dates.DatesRange)  as  Calendar_Quarter
,
case 
when month(dates.DatesRange) = 1 then 4 
when month(dates.DatesRange) = 2 then 5 
when month(dates.DatesRange) = 3 then 6 
when month(dates.DatesRange) = 4 then 7 
when month(dates.DatesRange) = 5 then 8 
when month(dates.DatesRange) = 6 then 9 
when month(dates.DatesRange) = 7 then 10 
when month(dates.DatesRange) = 8 then 11 
when month(dates.DatesRange) = 9 then 12
when month(dates.DatesRange) = 10 then 1 
when month(dates.DatesRange) = 11 then 2 
when month(dates.DatesRange) = 12 then 3 

End as Fiscal_Month
,
case 
when month(dates.DatesRange) in (1,2,3,4,5,6,7,8,9) then PERIOD_ADD(EXTRACT(YEAR_MONTH FROM dates.DatesRange),-12) +3 
when month(dates.DatesRange) in (10,11,12) then PERIOD_ADD(EXTRACT(YEAR_MONTH FROM dates.DatesRange),-9)

End as Fiscal_Yrmo
,
case 
when month(dates.DatesRange) in (1,2,3) then 2 
when month(dates.DatesRange) in (4,5,6) then 3  
when month(dates.DatesRange) in (7,8,9) then 4  
when month(dates.DatesRange) in (10,11,12) then 1  

End as Fiscal_Quarter
,
case 
when month(dates.DatesRange) in (1,2,3,4,5,6,7,8,9) then (year(dates.DatesRange)-1)  
when month(dates.DatesRange) in (10,11,12) then year(dates.DatesRange)

End as Fiscal_Year


,true as IsActive
,now() as Added_Date 
,'ETL_Admin' as Added_User
,now() as Updated_Date
,'ETL_Admin' as Updated_User

From
(
SELECT '1990-01-01' + INTERVAL (rank - 1) DAY  as DatesRange
FROM ( SELECT @rownum := @rownum + 1 AS rank    FROM INFORMATION_SCHEMA.STATISTICS t, (SELECT @rownum := 0) r ) a
WHERE rank BETWEEN 1 AND (365*100)   
)  dates ;
