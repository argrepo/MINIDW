   metastore.DBS d,

   metastore.COLUMNS_V2 c,
   metastore.CDS cds,
   metastore.KEY_CONSTRAINTS kc,
   metastore.DBS p_d,
   metastore.TBLS p_t,
   metastore.SDS p_s,
   metastore.COLUMNS_V2 p_c,
   metastore.CDS p_cds

   d.DB_ID = t.DB_ID AND t.SD_ID = s.SD_ID
       AND s.cd_id = c.cd_id
       AND cds.CD_ID = c.CD_ID
       AND kc.CHILD_TBL_ID = t.TBL_ID
       AND kc.CHILD_CD_ID = c.CD_ID

       AND p_d.DB_ID = p_t.DB_ID

public int db2DataExtraction(DataExtractionTask dataExtractionTask,MessagingConnectionFactory messagingConnectionFactory) throws ClassNotFoundException, SQLException
	{
		int objectsExtracted = 0;
		ResultSet resultSet = null;
		Connection connection = null;
		PreparedStatement preparedStatement = null;
		
		try ( final MessagingConnection mc = messagingConnectionFactory.connect())
		{

			MessagingQueue outputMessagingQueue = mc.queue("DES-"+dataExtractionTask.getJobId());
			
			Map<String, String> contextParams = dataExtractionTask.getContextParameters();

			connection = getConnection(contextParams);

			if( connection != null )
			{

				String columns[] = contextParams.get("DATASOURCE_COLUMN_NAMES").split("@#@");

				StringJoiner dataSourceColumns = new StringJoiner(",");

				for (int i = 0; i < columns.length; i++)
				{
					dataSourceColumns.add("`" + columns[i] + "`");
				}

				String tableName = "`" + contextParams.get("DATASOURCE_TABLE_NAME") + "`";

				String databaseName = "`" + contextParams.get("DATASOURCE_NAME") + "`";

				String query = "Select " + dataSourceColumns + " from " + databaseName + "." + tableName + " LIMIT " + contextParams.get("LIMIT") + " OFFSET " + contextParams.get("OFFSET");

				preparedStatement = connection.prepareStatement(query,ResultSet.TYPE_SCROLL_SENSITIVE,ResultSet.CONCUR_READ_ONLY );
			 
				preparedStatement.setFetchSize(500);
				 
				resultSet = preparedStatement.executeQuery();

				while (resultSet.next())
				{
					final JsonObjectBuilder jsonObjectBuilder = Json.createObjectBuilder();
					
					for (int i = 0; i < columns.length; i++)
					{
						final String name = columns[i];
						final Object value = resultSet.getObject(name);
						if(value != null)
						{
							jsonObjectBuilder.add(name, value.toString());
						}
					}
					
				    outputMessagingQueue.post(jsonObjectBuilder.build().toString());
					 
					objectsExtracted++;
				}
				
			}
		}
		catch ( Exception e )
		{
			e.printStackTrace();
		}
		finally
		{
			if( preparedStatement != null )
			{
				preparedStatement.close();
			}
			if( resultSet != null )
			{
				resultSet.close();
			}
			if( connection != null )
			{
				connection.close();
			}

		}
		return objectsExtracted;
	}

    private static String getIdFromConflict(String message) {
        String id = "";
        Pattern p = Pattern.compile("\"id\":\"[0-9]+\"");
        Pattern p2 = Pattern.compile("[0-9]+");
        Matcher m = p.matcher(message);
        if (m.find()) {
            String sub = m.group();
            Matcher m2 = p2.matcher(sub);
            if (m2.find()) {
                id = m2.group();
            }
        }
        return id;
